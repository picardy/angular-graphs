Angular D3 Chart Directive

install docs github
Install via Bower
bower install angular-graphs --save
HTML
<d3-graph-pie render="drawPie"></d3-graph-pie>
Javascript
$scope.drawPie({
  colors: {
    slices: ['#F2CFDA', '#8A8FC0'],
    amount: '#222'
  },
  start: [100, 0],
  end: [20, 80]
});
Javascript
$scope.drawPie({
  labels: ['Incomplete', 'Complete'],
  colors: {
    slices: ['#F2CFDA', '#8A8FC0'],
    amount: '#222',
    lines: '#666',
    labels: '#222'
  },
  start: [100, 0],
  end: [20, 80]
});
